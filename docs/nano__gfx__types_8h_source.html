205  inline SSD1306_RECT getUpdateRect() const
206  {
207  uint8_t left = min(x,lx);
208  uint8_t top = min(y,ly);
209  uint8_t right = max((uint8_t)(x + w - 1), (uint8_t)(lx + w - 1));
210  if (((uint8_t)(lx + w - 1) < w) && (right > 2*w))
211  {
212  right = (uint8_t)(lx + w - 1);
213  }
214  uint8_t bottom = max((uint8_t)(y + 7), (uint8_t)(ly + 7));
215  if (((uint8_t)(ly + 7) < 8) && (bottom > 16))
216  {
217  bottom = (uint8_t)(ly + 7);
218  }
219  if ( left > right ) left = 0;
220  if ( top > bottom ) top = 0;
221  return (SSD1306_RECT){ left, top, right, bottom };
