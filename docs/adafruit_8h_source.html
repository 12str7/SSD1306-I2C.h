97 template <>
98 void AdafruitCanvasOps<1>::drawPixel(int16_t x, int16_t y, uint16_t color)
99 {
100  x -= offset.x;
101  y -= offset.y;
102  if ((x < 0) || (x >= width()) || (y < 0) || (y >= height()))
103  {
104  return;
105  }
106 
107  switch (getRotation()) {
108  case 1:
109  ssd1306_swap_data(x, y, int16_t);
110  x = width() - x - 1;
111  break;
112  case 2:
113  x = width() - x - 1;
114  y = height() - y - 1;
115  break;
116  case 3:
117  ssd1306_swap_data(x, y, int16_t);
118  y = height() - y - 1;
119  break;
120  }
121 
122  switch (color)
123  {
124  case 1: m_buffer[x+ (y/8)*width()] |= (1 << (y&7)); break;
125  case 0: m_buffer[x+ (y/8)*width()] &= ~(1 << (y&7)); break;
126  case 2: m_buffer[x+ (y/8)*width()] ^= (1 << (y&7)); break;
127  }
128 }
