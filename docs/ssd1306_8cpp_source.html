75 {
76  ssd1306_i2cStart();
77  ssd1306_i2cSendByte(0x00); //write command
78  ssd1306_i2cSendByte(0xb0+y);

85 void ssd1306_fillScreen(uint8_t fill_Data){
86  uint8_t m,n;
87  for(m=0;m<8;m++)
88  {
89  ssd1306_i2cSendCommand(0xb0+m); //page0-page1
90  ssd1306_i2cSendCommand(0x00); //low column start address
91  ssd1306_i2cSendCommand(0x10); //high column start address
92  ssd1306_i2cDataStart();
93  for(n=0;n<128;n++)
94  {
95  ssd1306_i2cSendByte(fill_Data);
96  }
97  ssd1306_i2cStop();
98  }

128 void ssd1306_charF6x8(uint8_t x, uint8_t y, const char ch[]){
129  uint8_t c,i,j=0;
130  while(ch[j] != '\0')
131  {
132  c = ch[j] - 32;
133  if(x>126)
134  {
135  x=0;
136  y++;
137  }
138  ssd1306_setPos(x,y);
139  ssd1306_i2cDataStart();
140  for(i=0;i<6;i++)
141  {
142  ssd1306_i2cSendByte(pgm_read_byte(&ssd1306xled_font6x8[c*6+i]));
143  }
144  ssd1306_i2cStop();
145  x += 6;
146  j++;
147  }
148 }

195 void ssd1306_drawSprite(SPRITE *sprite)
196 {
197  uint8_t posy = sprite->y >> 3;
198  uint8_t offsety = sprite->y & 0x7;
199  ssd1306_setPos(sprite->x, posy);
200  ssd1306_i2cDataStart();
201  for (uint8_t i=0; i < sprite->w; i++)
202  {
203  ssd1306_i2cSendByte( pgm_read_byte( &sprite->data[i] ) << offsety );
204  }
205  ssd1306_i2cStop();
206  if (offsety)
207  {
208  ssd1306_setPos(sprite->x, posy + 1);
209  ssd1306_i2cDataStart();
210  for (uint8_t i=0; i < sprite->w; i++)
211  {
212  ssd1306_i2cSendByte( pgm_read_byte( &sprite->data[i] ) >> (8 - offsety) );
213  }
214  }
215  ssd1306_i2cStop();
216  sprite->lx = sprite->x;
217  sprite->ly = sprite->y;
218 }

245 void ssd1306_eraseTrace(SPRITE *sprite)
246 {
247  uint8_t y1 = sprite->ly >> 3;
248  uint8_t y2 = (sprite->ly + 7) >> 3;
249  if (sprite->ly < sprite->y)
250  y2 = min(y2, (uint8_t)((sprite->y >> 3) - 1));
251  else
252  y1 = max(y1, (sprite->ly + 7) >> 3);
253  for(uint8_t y = y1; y <= y2; y++)
254  {
255  ssd1306_setPos(sprite->lx, y);
256  ssd1306_i2cDataStart();
257  for(uint8_t x = sprite->lx; x < sprite->lx + sprite->w; x++)
258  {
259  ssd1306_i2cSendByte( B00000000 );
260  }
261  ssd1306_i2cStop();
262  }
263  if (sprite->lx != sprite->x)
264  {
265  uint8_t x1 = sprite->lx;
266  uint8_t x2 = sprite->lx + sprite->w - 1;
267  if (sprite->x < sprite->lx)
268  x1 = max(x1, sprite->x + sprite->w);
269  else
270  x2 = min((uint8_t)(sprite->x - 1), x2);
271  for(uint8_t y = sprite->ly >> 3; y <= (sprite->ly + 7) >> 3; y++)
272  {
273  ssd1306_setPos(x1, y);
274  ssd1306_i2cDataStart();
275  for(uint8_t x = x1; x <= x2; x++)
276  {
277  ssd1306_i2cSendByte( B00000000 );
278  }
279  ssd1306_i2cStop();
280  }
281  }
282 }
