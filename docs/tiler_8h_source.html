208  {
209  if ((point.x<0)||(point.y<0) || ((point.y/canvas.height())>=NE_MAX_TILES_NUM)) return;
210  m_refreshFlags[(point.y/canvas.height())] |= (1<<(point.x/canvas.width()));

217  void refresh(lcdint_t x1, lcdint_t y1, lcdint_t x2, lcdint_t y2)
218  {
219  if (y2 < 0 || x2 < 0) return;
220  if (y1 < 0) y1 = 0;
221  if (x1 < 0) x1 = 0;
222  y1 = y1/canvas.height();
223  y2 = min((y2/canvas.height()), NE_MAX_TILES_NUM - 1);
224  for (uint8_t y=y1; y<=y2; y++)
225  {
226  for(uint8_t x=x1/canvas.width(); x<=(x2/canvas.width()); x++)
227  {
228  m_refreshFlags[y] |= (1<<x);
229  }
230  }
231  }

461 template<class C, uint8_t B>
462 void NanoEngineTiler<C,B>::displayBuffer()
463 {
464  for (lcduint_t y = 0; y < m_display.height(); y = y + canvas.height())
465  {
466  uint16_t flag = m_refreshFlags[y/canvas.height()];
467  m_refreshFlags[y/canvas.height()] = 0;
468  for (lcduint_t x = 0; x < m_display.width(); x = x + canvas.width())
469  {
470  if (flag & 0x01)
471  {
472  canvas.setOffset(x + offset.x, y + offset.y);
473  if (!m_onDraw)
474  {
475  canvas.clear();
476  draw();
477  m_display.drawCanvas(x,y,canvas);
478  }
479  else if (m_onDraw())
480  {
481  draw();
482  m_display.drawCanvas(x,y,canvas);
483  }
484  }
485  flag >>=1;
486  }
487  }
488 }

491 void NanoEngineTiler<C,B>::displayPopup(const char *msg)
492 {
493  NanoRect rect = { {8, (m_display.height()>>1) - 8}, {m_display.width() - 8, (m_display.height()>>1) + 8} };
494  // TODO: It would be nice to calculate message height
495  NanoPoint textPos = { (m_display.width() - (lcdint_t)strlen(msg)*s_fixedFont.h.width) >> 1, (m_display.height()>>1) - 4 };
496  refresh(rect);
497  for (lcduint_t y = 0; y < m_display.height(); y = y + canvas.height())
498  {
499  uint16_t flag = m_refreshFlags[y/canvas.height()];
500  m_refreshFlags[y/canvas.height()] = 0;
501  for (lcduint_t x = 0; x < m_display.width(); x = x + canvas.width())
502  {
503  if (flag & 0x01)
504  {
505  canvas.setOffset(x + offset.x, y + offset.y);
506  if (!m_onDraw)
507  {
508  canvas.clear();
509  draw();
510  }
511  else if (m_onDraw())
512  {
513  draw();
514  }
515  canvas.setOffset(x, y);
516  canvas.setColor(RGB_COLOR8(0,0,0));
517  canvas.fillRect(rect);
518  canvas.setColor(RGB_COLOR8(192,192,192));
519  canvas.drawRect(rect);
520  canvas.printFixed( textPos.x, textPos.y, msg);
521 
522  m_display.drawCanvas(x,y,canvas);
523  }
524  flag >>=1;
525  }
526  }
