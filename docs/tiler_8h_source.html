119  static void refresh(const NanoPoint &point)
120  {
121  if ((point.y<0) || ((point.y>>B)>=NE_MAX_TILES_NUM)) return;
122  m_refreshFlags[(point.y>>B)] |= (1<<(point.x>>B));

129  static void refresh(lcdint_t x1, lcdint_t y1, lcdint_t x2, lcdint_t y2)
130  {
131  if (y2 < 0) return;
132  if (y1 < 0) y1 = 0;
133  if (x1 < 0) x1 = 0;
134  y1 = y1>>B;
135  y2 = min((y2>>B), NE_MAX_TILES_NUM - 1);
136  for (uint8_t y=y1; y<=y2; y++)
137  {
138  for(uint8_t x=x1>>B; x<=(x2>>B); x++)
139  {
140  m_refreshFlags[y] |= (1<<x);
141  }
142  }
143  }

286 template<class C, uint8_t W, uint8_t H, uint8_t B>
287 void NanoEngineTiler<C,W,H,B>::displayBuffer()
288 {
289  if (!m_onDraw) // If onDraw handler is not set, just output current canvas
290  {
291  canvas.blt();
292  return;
293  }
294  for (uint8_t y = 0; y < ssd1306_lcd.height; y = y + NE_TILE_HEIGHT)
295  {
296  uint16_t flag = m_refreshFlags[y >> NE_TILE_SIZE_BITS];
297  m_refreshFlags[y >> NE_TILE_SIZE_BITS] = 0;
298  for (uint8_t x = 0; x < ssd1306_lcd.width; x = x + NE_TILE_WIDTH)
299  {
300  if (flag & 0x01)
301  {
302  canvas.setOffset(x, y);
303  if (m_onDraw())
304  {
305  canvas.setOffset(x, y);
306  canvas.blt();
307  }
308  }
309  flag >>=1;
310  }
311  }
312 }

314 template<class C, uint8_t W, uint8_t H, uint8_t B>
315 void NanoEngineTiler<C,W,H,B>::displayPopup(const char *msg)
316 {
317  NanoRect rect = { {8, (ssd1306_lcd.height>>1) - 8}, {ssd1306_lcd.width - 8, (ssd1306_lcd.height>>1) + 8} };
318  // TODO: It would be nice to calculate message height
319  NanoPoint textPos = { (ssd1306_lcd.width - (lcdint_t)strlen(msg)*s_fixedFont.width) >> 1, (ssd1306_lcd.height>>1) - 4 };
320  refresh(rect);
321  for (uint8_t y = 0; y < ssd1306_lcd.height; y = y + NE_TILE_HEIGHT)
322  {
323  uint16_t flag = m_refreshFlags[y >> NE_TILE_SIZE_BITS];
324  m_refreshFlags[y >> NE_TILE_SIZE_BITS] = 0;
325  for (uint8_t x = 0; x < ssd1306_lcd.width; x = x + NE_TILE_WIDTH)
326  {
327  if (flag & 0x01)
328  {
329  canvas.setOffset(x, y);
330  if (m_onDraw) m_onDraw();
331  canvas.setOffset(x, y);
332  canvas.setColor(RGB_COLOR8(0,0,0));
333  canvas.fillRect(rect);
334  canvas.setColor(RGB_COLOR8(192,192,192));
335  canvas.drawRect(rect);
336  canvas.printFixed( textPos.x, textPos.y, msg);
337 
338  canvas.blt();
339  }
340  flag >>=1;
341  }
342  }
343 }
